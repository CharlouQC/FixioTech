import { Navigate, Outlet } from "react-router-dom"; import { useAuth } from "../context/AuthContext"; export default function ProtectedRoute({ allowedRoles }) { const { isAuthenticated, role } = useAuth(); if (!isAuthenticated) return <Navigate to="/login" replace />; if (allowedRoles && !allowedRoles.includes(role)) { // Option: route 403 dédiée return <Navigate to="/" replace />; } return <Outlet />; }
