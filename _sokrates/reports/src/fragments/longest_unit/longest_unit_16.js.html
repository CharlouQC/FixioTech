async function loginUtilisateur(utilisateur) {
  const payload = {
    email: utilisateur.courriel,
    mot_de_passe: utilisateur.mot_de_passe,
  };

  return httpJson(`${API_URL}/login`, {
    method: "POST",
    body: JSON.stringify(payload),
  });
}
